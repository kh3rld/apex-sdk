Quick Example

use apex_sdk::prelude::*;

#[tokio::main]
async fn main() -> Result<()> {
    // Initialize SDK with unified configuration
    let sdk = ApexSDK::builder()
        .with_substrate_endpoint("wss://polkadot.api.onfinality.io/public-ws")
        .with_evm_endpoint("https://mainnet.infura.io/v3/YOUR_KEY")
        .build()
        .await?;

    // Create a cross-chain transaction with compile-time type safety
    let tx = sdk
        .transaction()
        .from_substrate_account(Ilara)
        .to_evm_address("0x742d35Cc6634C0532925a3b844Bc9e7595f0bEb7")
        .amount(1000_u128)
        .build()?;

    // Execute with automatic routing
    let result = sdk.execute(tx).await?;
    println!("Transaction successful: {:?}", result);

    Ok(())
}
