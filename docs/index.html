Build cross-chain applications with just a few lines of code. Apex SDK handles the complexity while providing compile-time guarantees.

  • Automatic chain detection and routing
  • Built-in wallet management
  • Transaction fee optimization
  • Error handling and retries
  • Real-time event subscriptions
  • Comprehensive testing framework
src/main.rs 
use apex_sdk::prelude::*;

// Cross-chain transfer in just a few lines
#[tokio::main]
async fn main() -> Result<()> {
    // Initialize with multiple chains
    let sdk = ApexSDK::builder()
        .with_substrate_endpoint("wss://paseo-rpc.polkadot.io")
        .with_evm_endpoint("https://eth.llamarpc.com")
        .build().await?;

    // Create cross-chain transaction
    let tx = sdk.transaction()
        .from_substrate_account(ALICE)
        .to_evm_address("0x742d35...b9e7595f0bEb7")
        .amount(1000_u128)
        .build()?;

    // Execute with automatic routing
    let result = sdk.execute(tx).await?;
    println!("Success: {:?}", result);
    
    Ok(())
}
