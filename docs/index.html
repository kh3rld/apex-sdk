Quick Start

Start Building in Minutes

Get up and running with just a few lines of code. Here's a complete cross-chain transaction example. 

use apex_sdk::prelude::*;

// Initialize SDK with unified configuration
let sdk = ApexSDK::builder()
    .with_substrate_endpoint("wss://paseo.rpc.amforc.com")
    .with_evm_endpoint("https://eth-mainnet.g.alchemy.com/v2/YOUR_KEY")
    .build()
    .await?;

// Create a cross-chain transaction
let tx = sdk
    .transaction()
    .from_substrate_account("Alice")
    .to_evm_address("0x742d35Cc6634C0532925a3b844Bc9e7595f0bEb7")
    .amount(1000_u128)
    .with_nonce(1)
    .build()?;

// Execute with automatic routing and deterministic hashing
let result = sdk.execute(tx).await?;
println!("Transaction hash: {}", result.hash());
